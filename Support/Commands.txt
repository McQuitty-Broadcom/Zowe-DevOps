# 
# Create your profiles
#
zowe files list ds MAINFRAME_USER.*

#
# Verify Setup
#
zowe zosmf check status
zowe files list ds "MAINFRAME_USER.MARBLES.*"

#
# Validate Endevor Profile
#
zowe endevor list elements

#
# Update Endevor Elements
#
zowe endevor ret element MARBLE_NUM --type COBOL --tf MARBLE_NUM.cbl --sn 1

#
# Edit the code in Che, then update it
#
zowe endevor update element MARBLE_NUM --type COBOL --ff MARBLE_NUM.cbl --os

#
# Generate the Elements manually
#
zowe endevor generate element MARBLE_NUM --type COBOL --sn 1
zowe endevor generate element MARBLE_NUM --type LNK --sn 1

#
# Start Using Gulp.
# 
# gulp will list the available commands
#
gulp build-cobol
gulp build-lnk

#
# After combining the gulp build-cobol and build-lnk tasts
#
gulp build

#
# After completing gulp deploy tasks
#
gulp deploy

#
# Manually verify
#
zowe console issue command "F CICS_REGION,MARBLE_CICS CRE MARBLE_COLOR 1 2" --cn MAINFRAME_USER
zowe jobs submit ds "MAINFRAME_USER.MARBLES.JCL(MARBDB2)" --vasc 

# 
# Run automated tests
# 
npm test
